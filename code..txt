<script>

const words=[
"你好",
"谢谢",
"再见",
"老师",
"学生",
"中国",
"喜欢",
"今天",
"朋友",
"学习",
"我想喝水",
"你吃饭了吗",
"今天天气很好",
"我喜欢中国菜",
"欢迎来到中国",
"请慢一点说",
"你叫什么名字",
"我不会说中文",
"我们一起学习",
"祝你好运"
];

let questions=[];
let current=0;
let scores=[];

function showPackage(){

welcome.classList.remove("active");
package.classList.add("active");

}

function shuffle(arr){

return [...arr].sort(()=>Math.random()-0.5);

}

function start(type){

questions=shuffle(words).slice(0,10);

package.classList.remove("active");

learn.classList.add("active");

showQuestion();

}

function showQuestion(){

progress.innerHTML=`Câu ${current+1}/10`;

word.innerHTML=questions[current];

feedback.innerHTML="";

}

function record(){

alert("🎤 Đang ghi âm (Demo)");

}

function submit(){

let score=Math.floor(Math.random()*21)+80;

scores.push(score);

let text="";

if(score>=95){

text="🌟 Hoàn hảo!";

}

else if(score>=90){

text="😊 Rất tốt!";

}

else{

text="👍 Khá tốt, cần luyện thêm thanh điệu.";

}

feedback.innerHTML=`<h2>${score}/100</h2>${text}`;

setTimeout(()=>{

current++;

if(current==10){

finish();

}else{

showQuestion();

}

},1500);

}

function finish(){

learn.classList.remove("active");

finish.classList.add("active");

let avg=(scores.reduce((a,b)=>a+b)/scores.length).toFixed(1);

document.getElementById("avg").innerHTML="Điểm trung bình: "+avg+"/100";

}

</script>
